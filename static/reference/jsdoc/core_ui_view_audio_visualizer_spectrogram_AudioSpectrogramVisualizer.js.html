    render() {
        const time = performance.now();
        if(time - this.lastTime > this.clearThreshold) {
            this.DATA = null;

    update_geometry() {
        let start_val = this.frequency_samples + 1;
        let end_val = this.n_vertices - start_val;
        this.heights.copyWithin(0, start_val, this.n_vertices + 1);
        this.heights.set(this.DATA, end_val - start_val);
        this.mesh.geometry.setAttribute('displacement', new THREE.Uint8BufferAttribute(this.heights, 1));
    }

    draw(decodedSample) {
        this.DATA =  decodedSample[this.properties.type][this.properties.format];
        this.lastTime = performance.now();
    }

    onended() {
        this.lastTime = performance.now();
    }
